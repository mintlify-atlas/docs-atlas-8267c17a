## Overview

The `app_list_tickets` endpoint returns all tickets owned by a user, including ticket details, event information, QR codes, and access status.

## Request

### Endpoint

```
POST https://us-central1-{project}.cloudfunctions.net/app_list_tickets
```

### Headers

| Header | Value |
|--------|-------|
| Content-Type | `application/json` |
| Authorization | `Bearer {firebase_id_token}` |

### Body Parameters

<ParamField path="user_id" type="string" required>
  User ID from Firebase Authentication
</ParamField>

<ParamField path="event_id" type="string">
  Filter tickets by specific event ID (optional)
</ParamField>

<ParamField path="status" type="string">
  Filter by access status: `"unused"`, `"used"`, or `"all"` (default)
</ParamField>

```javascript JavaScript
const listTickets = async (userId, eventId = null, status = 'all') => {
  const response = await fetch(
    'https://us-central1-{project}.cloudfunctions.net/app_list_tickets',
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${idToken}`
      },
      body: JSON.stringify({
        data: { user_id: userId, event_id: eventId, status }
      })
    }
  );
  
  return await response.json();
};

const tickets = await listTickets('user_abc123', 'evt_xyz789', 'unused');
console.log(tickets);
```

```swift iOS (Swift)
import FirebaseFunctions

func listTickets(userId: String, eventId: String? = nil) async throws -> [Ticket] {
    let functions = Functions.functions()
    var data: [String: Any] = ["user_id": userId]
    if let eventId = eventId {
        data["event_id"] = eventId
    }
    
    let result = try await functions.httpsCallable("app_list_tickets").call(data)
    guard let tickets = result.data as? [[String: Any]] else {
        throw TicketError.invalidResponse
    }
    
    return tickets.compactMap { Ticket(dictionary: $0) }
}

let tickets = try await listTickets(userId: user.uid, eventId: "evt_xyz789")
```

## Response

### Success Response

<ResponseField name="status" type="number">
  HTTP status code (200 for success)
</ResponseField>

<ResponseField name="data" type="array">
  Array of ticket objects
  
  <Expandable title="Ticket object">
    <ResponseField name="ticket_id" type="string">
      Unique ticket identifier
    </ResponseField>
    
    <ResponseField name="event_id" type="string">
      Associated event ID
    </ResponseField>
    
    <ResponseField name="event_name" type="string">
      Name of the event
    </ResponseField>
    
    <ResponseField name="event_date" type="timestamp">
      Event start date and time
    </ResponseField>
    
    <ResponseField name="zone" type="string">
      Seating zone name
    </ResponseField>
    
    <ResponseField name="seat_id" type="string">
      Seat identifier (e.g., "VIP-1")
    </ResponseField>
    
    <ResponseField name="seat_row" type="string">
      Row identifier
    </ResponseField>
    
    <ResponseField name="color" type="string">
      Zone color code (hex)
    </ResponseField>
    
    <ResponseField name="price" type="number">
      Ticket price
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      Currency code
    </ResponseField>
    
    <ResponseField name="qr_code" type="string">
      Base64 encoded QR code image or QR data string
    </ResponseField>
    
    <ResponseField name="access_status" type="boolean">
      Whether ticket has been used to enter venue
    </ResponseField>
    
    <ResponseField name="access_entry" type="timestamp" optional>
      Entry timestamp if accessed
    </ResponseField>
    
    <ResponseField name="access_exit" type="timestamp" optional>
      Exit timestamp if customer left venue
    </ResponseField>
    
    <ResponseField name="transferable" type="boolean">
      Whether ticket can be transferred to another user
    </ResponseField>
  </Expandable>
</ResponseField>

## Offline Support

<Tip>
  Cache tickets locally for offline access:
</Tip>

```javascript
import AsyncStorage from '@react-native-async-storage/async-storage';

const cacheTickets = async (tickets) => {
  await AsyncStorage.setItem('cached_tickets', JSON.stringify(tickets));
};

const getCachedTickets = async () => {
  const cached = await AsyncStorage.getItem('cached_tickets');
  return cached ? JSON.parse(cached) : [];
};

// Load tickets with offline fallback
const loadTickets = async (userId) => {
  try {
    const result = await listTickets(userId);
    await cacheTickets(result.data);
    return result.data;
  } catch (error) {
    console.warn('Using cached tickets:', error);
    return await getCachedTickets();
  }
};
```
