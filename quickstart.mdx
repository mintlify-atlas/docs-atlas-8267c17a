## Prerequisites

Before you begin, ensure you have:

- A Firebase project with Cloud Functions enabled
- Node.js 22 or higher installed
- Firebase Admin SDK credentials
- An API endpoint URL (deployed Cloud Functions URL)

<Note>
  TMT API is built on Firebase Cloud Functions v2. All endpoints are HTTPS callable functions that require Firebase Authentication.
</Note>

## Installation

<Steps>
  <Step title="Install required dependencies">
    Install the necessary Firebase and HTTP client libraries:

    <CodeGroup>
      ```bash npm
      npm install firebase-admin cors axios
      ```

      ```bash yarn
      yarn add firebase-admin cors axios
      ```
    </CodeGroup>
  </Step>

  <Step title="Configure Firebase Admin">
    Initialize Firebase Admin SDK with your service account credentials:

    ```javascript
    const admin = require('firebase-admin');
    const serviceAccount = require('./path/to/serviceAccountKey.json');

    admin.initializeApp({
      credential: admin.credential.cert(serviceAccount),
      projectId: 'your-project-id'
    });

    const db = admin.firestore();
    const auth = admin.auth();
    ```

    <Warning>
      Keep your service account key secure. Never commit it to version control or expose it publicly.
    </Warning>
  </Step>

  <Step title="Set your API base URL">
    Define the base URL for your deployed Cloud Functions:

    ```javascript
    const API_BASE_URL = 'https://us-central1-your-project-id.cloudfunctions.net';
    ```
  </Step>
</Steps>

### Example 2: List Event Tickets

Query all tickets for a specific event using the `tickets_list_event` endpoint.

<CodeGroup>
  ```javascript Node.js
  async function listEventTickets(eventId) {
    try {
      const response = await axios.post(
        `${API_BASE_URL}/tickets_list_event`,
        {
          data: {
            event_id: eventId
          }
        },
        {
          headers: {
            'Content-Type': 'application/json'
          }
        }
      );

      console.log('Tickets found:', response.data.data.response.length);
      return response.data.data.response;
    } catch (error) {
      console.error('Error fetching tickets:', error.response?.data || error.message);
      throw error;
    }
  }

  // Usage
  listEventTickets('your-event-id-here')
    .then(tickets => {
      console.log('First ticket:', tickets[0]);
    });
  ```

  ```python Python
  def list_event_tickets(event_id):
      url = f"{API_BASE_URL}/tickets_list_event"
      
      payload = {
          "data": {
              "event_id": event_id
          }
      }
      
      headers = {
          "Content-Type": "application/json"
      }
      
      try:
          response = requests.post(url, json=payload, headers=headers)
          response.raise_for_status()
          
          tickets = response.json()['data']['response']
          print(f"Tickets found: {len(tickets)}")
          
          return tickets
      except requests.exceptions.RequestException as e:
          print(f"Error fetching tickets: {e}")
          raise

  # Usage
  if __name__ == "__main__":
      tickets = list_event_tickets('your-event-id-here')
      if tickets:
          print(f"First ticket: {tickets[0]}")
  ```

## Complete Working Example

Here's a full example that demonstrates creating a client, validating their user type, and fetching event tickets:

```javascript
const axios = require('axios');

const API_BASE_URL = 'https://us-central1-your-project-id.cloudfunctions.net';

class TMTApiClient {
  constructor(baseUrl) {
    this.baseUrl = baseUrl;
    this.axios = axios.create({
      baseURL: baseUrl,
      headers: {
        'Content-Type': 'application/json'
      }
    });
  }

  async createClient(userData) {
    try {
      const response = await this.axios.post('/create_client', {
        data: userData
      });
      return response.data;
    } catch (error) {
      throw new Error(`Failed to create client: ${error.response?.data?.message || error.message}`);
    }
  }

  async validateUserType(email, platform, accountType) {
    try {
      const response = await this.axios.post('/validate_user_type', {
        data: {
          email: email,
          platform: platform,
          account_type: accountType
        }
      });
      return response.data;
    } catch (error) {
      throw new Error(`Failed to validate user: ${error.response?.data?.message || error.message}`);
    }
  }

  async listEventTickets(eventId) {
    try {
      const response = await this.axios.post('/tickets_list_event', {
        data: { event_id: eventId }
      });
      return response.data;
    } catch (error) {
      throw new Error(`Failed to list tickets: ${error.response?.data?.message || error.message}`);
    }
  }

  async lockTickets(ticketIds, reservedTime = 600) {
    try {
      const response = await this.axios.post('/tickets_lock', {
        data: {
          ticketd_reserved: ticketIds,
          reserved_time: reservedTime
        }
      });
      return response.data;
    } catch (error) {
      throw new Error(`Failed to lock tickets: ${error.response?.data?.message || error.message}`);
    }
  }
}

// Usage example
async function main() {
  const client = new TMTApiClient(API_BASE_URL);

  try {
    // 1. Create a new client
    console.log('Creating client...');
    const newClient = await client.createClient({
      name: 'Alice Smith',
      email: 'alice@example.com',
      password: 'SecurePass123!',
      name_commercial: 'Alice Events LLC',
      phone: '+1234567890',
      status: true,
      id: '87654321',
      id_type: 'national_id',
      city: 'San Francisco',
      country: 'USA',
      line: '456 Market St',
      zip_code: '94102',
      account_type: 'business',
      commercial_registry: 'REG-2024-002',
      last_assembly_minutes: '',
      fiscal_info: 'EIN: 98-7654321',
      commercial_license: 'LIC-2024-002',
      taxpayer_type: 'corporate',
      legal_representative: 'Alice Smith',
      contact_person: 'Bob Johnson',
      bank_accounts: []
    });
    console.log('✓ Client created:', newClient.data.uid);

    // 2. Validate user type
    console.log('\nValidating user...');
    const validation = await client.validateUserType(
      'alice@example.com',
      'u_clients',
      'business'
    );
    console.log('✓ User validated:', validation.data.valido);

    // 3. List tickets for an event
    console.log('\nFetching event tickets...');
    const tickets = await client.listEventTickets('your-event-id');
    console.log(`✓ Found ${tickets.data.response.length} tickets`);

    // 4. Lock tickets for purchase
    if (tickets.data.response.length > 0) {
      const ticketToLock = tickets.data.response[0].ticket_id;
      console.log('\nLocking ticket...');
      const locked = await client.lockTickets([ticketToLock], 300);
      console.log('✓ Ticket locked:', locked.data.valido);
    }

  } catch (error) {
    console.error('Error:', error.message);
  }
}

main();
```
