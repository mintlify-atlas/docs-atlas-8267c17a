## Overview

TMT API uses **Firebase Authentication** to secure all Cloud Functions endpoints. Every request must include a valid Firebase ID token to authenticate the user and authorize access to protected resources.

<Note>
  Firebase Authentication provides enterprise-grade security with built-in user management, token validation, and session handling.
</Note>

## How Authentication Works

The TMT API authentication flow consists of three key steps:

<Steps>
  <Step title="User Authentication">
    Users authenticate with Firebase using email/password, OAuth providers, or custom tokens.
  </Step>
  <Step title="Token Generation">
    Firebase generates a signed JWT (JSON Web Token) called an ID token that contains the user's identity.
  </Step>
  <Step title="Request Authorization">
    The client includes the ID token in API requests, and Cloud Functions validate it before processing.
  </Step>
</Steps>

## Token Management

### Getting ID Tokens

Firebase ID tokens are short-lived (1 hour by default) and must be refreshed regularly:

```javascript
const { getAuth } = require('firebase/auth');

class TokenManager {
  constructor() {
    this.auth = getAuth();
    this.currentToken = null;
    this.tokenExpiry = null;
  }

  async getValidToken() {
    const user = this.auth.currentUser;
    
    if (!user) {
      throw new Error('User not authenticated');
    }

    // Check if token needs refresh (refresh 5 minutes before expiry)
    const now = Date.now();
    if (!this.currentToken || !this.tokenExpiry || now >= this.tokenExpiry - 5 * 60 * 1000) {
      console.log('Refreshing token...');
      this.currentToken = await user.getIdToken(true); // Force refresh
      
      // Tokens expire in 1 hour
      this.tokenExpiry = now + 60 * 60 * 1000;
    }

    return this.currentToken;
  }

  async decodeToken() {
    const token = await this.getValidToken();
    const tokenResult = await this.auth.currentUser.getIdTokenResult();
    
    return {
      token: token,
      claims: tokenResult.claims,
      expirationTime: tokenResult.expirationTime,
      authTime: tokenResult.authTime,
      issuedAtTime: tokenResult.issuedAtTime
    };
  }
}

// Usage
const tokenManager = new TokenManager();
const token = await tokenManager.getValidToken();
```

## Making Authenticated Requests

### Standard HTTP Requests

Include the ID token in the `Authorization` header with the `Bearer` scheme:

<CodeGroup>
  ```javascript Axios
  const axios = require('axios');

  class AuthenticatedApiClient {
    constructor(baseUrl, tokenManager) {
      this.baseUrl = baseUrl;
      this.tokenManager = tokenManager;
    }

    async makeRequest(endpoint, data) {
      const token = await this.tokenManager.getValidToken();
      
      try {
        const response = await axios.post(
          `${this.baseUrl}/${endpoint}`,
          { data },
          {
            headers: {
              'Content-Type': 'application/json',
              'Authorization': `Bearer ${token}`
            }
          }
        );
        
        return response.data;
      } catch (error) {
        if (error.response?.status === 401) {
          console.error('Authentication failed. Token may be invalid.');
        }
        throw error;
      }
    }
  }

  // Usage
  const apiClient = new AuthenticatedApiClient(
    'https://us-central1-your-project.cloudfunctions.net',
    tokenManager
  );

  const tickets = await apiClient.makeRequest('tickets_list_event', {
    event_id: 'event-123'
  });
  ```

  ```swift iOS (Keychain)
  import Security

  class KeychainManager {
      static func saveToken(_ token: String) {
          let data = token.data(using: .utf8)!
          let query: [String: Any] = [
              kSecClass as String: kSecClassGenericPassword,
              kSecAttrAccount as String: "firebase_token",
              kSecValueData as String: data
          ]
          
          SecItemDelete(query as CFDictionary)
          SecItemAdd(query as CFDictionary, nil)
      }
      
      static func getToken() -> String? {
          let query: [String: Any] = [
              kSecClass as String: kSecClassGenericPassword,
              kSecAttrAccount as String: "firebase_token",
              kSecReturnData as String: true
          ]
          
          var result: AnyObject?
          SecItemCopyMatching(query as CFDictionary, &result)
          
          if let data = result as? Data {
              return String(data: data, encoding: .utf8)
          }
          return nil
      }
  }
  ```

### 2. Token Refresh Strategy

```javascript
class SecureTokenManager {
  constructor(auth) {
    this.auth = auth;
    this.refreshTimer = null;
  }

  startAutoRefresh() {
    // Refresh token every 50 minutes (10 minutes before expiry)
    this.refreshTimer = setInterval(async () => {
      try {
        const user = this.auth.currentUser;
        if (user) {
          await user.getIdToken(true);
          console.log('Token refreshed automatically');
        }
      } catch (error) {
        console.error('Auto-refresh failed:', error);
      }
    }, 50 * 60 * 1000);
  }

  stopAutoRefresh() {
    if (this.refreshTimer) {
      clearInterval(this.refreshTimer);
      this.refreshTimer = null;
    }
  }
}

// Usage
const tokenManager = new SecureTokenManager(auth);
tokenManager.startAutoRefresh();

// Stop refresh on logout
auth.signOut().then(() => {
  tokenManager.stopAutoRefresh();
});
```

### 3. Error Handling

```javascript
class AuthError extends Error {
  constructor(code, message) {
    super(message);
    this.code = code;
    this.name = 'AuthError';
  }
}

async function handleAuthenticatedRequest(endpoint, data) {
  try {
    const token = await getValidToken();
    const response = await axios.post(endpoint, { data }, {
      headers: { 'Authorization': `Bearer ${token}` }
    });
    return response.data;
  } catch (error) {
    if (error.response?.status === 401) {
      throw new AuthError('INVALID_TOKEN', 'Authentication token is invalid or expired');
    } else if (error.response?.status === 403) {
      throw new AuthError('INSUFFICIENT_PERMISSIONS', 'User does not have required permissions');
    } else if (error.code === 'auth/network-request-failed') {
      throw new AuthError('NETWORK_ERROR', 'Network connection failed');
    }
    throw error;
  }
}
```

## Complete Authentication Example

Here's a full authentication flow implementation:

```javascript
const { initializeApp } = require('firebase/app');
const { getAuth, signInWithEmailAndPassword, signOut } = require('firebase/auth');
const axios = require('axios');

class TMTAuthenticatedClient {
  constructor(firebaseConfig, apiBaseUrl) {
    const app = initializeApp(firebaseConfig);
    this.auth = getAuth(app);
    this.apiBaseUrl = apiBaseUrl;
    this.currentToken = null;
  }

  async login(email, password) {
    try {
      const userCredential = await signInWithEmailAndPassword(
        this.auth,
        email,
        password
      );
      
      this.currentToken = await userCredential.user.getIdToken();
      console.log('✓ Logged in as:', userCredential.user.email);
      
      return {
        user: userCredential.user,
        token: this.currentToken
      };
    } catch (error) {
      console.error('Login failed:', error.code, error.message);
      throw error;
    }
  }

  async logout() {
    await signOut(this.auth);
    this.currentToken = null;
    console.log('✓ Logged out');
  }

  async refreshToken() {
    const user = this.auth.currentUser;
    if (!user) {
      throw new Error('No user logged in');
    }
    
    this.currentToken = await user.getIdToken(true);
    return this.currentToken;
  }

  async callApi(endpoint, data) {
    if (!this.currentToken) {
      throw new Error('Not authenticated. Please login first.');
    }

    try {
      const response = await axios.post(
        `${this.apiBaseUrl}/${endpoint}`,
        { data },
        {
          headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${this.currentToken}`
          }
        }
      );
      
      return response.data;
    } catch (error) {
      if (error.response?.status === 401) {
        console.log('Token expired, refreshing...');
        await this.refreshToken();
        return this.callApi(endpoint, data); // Retry
      }
      throw error;
    }
  }
}
```
